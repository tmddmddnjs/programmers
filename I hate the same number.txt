#include <vector>
#include <iostream>

using namespace std;

vector<int> solution(vector<int> arr){
	vector<int> answer;
	answer.push_back(arr[0]);
	for (int i = 1; i < arr.size(); i++) {
		if (arr[i-1] != arr[i]) {
			answer.push_back(arr[i]);
		}
	}
	for (int i = 0; i < answer.size(); i++) {
		cout << answer[i] << " ";
	}
	return answer;
}

int main() {
	vector<int> arr = { 1,1,3,3,0,1,1 }; //1301
	//vector<int> arr = {4,4,4,3,3}; //43
	solution(arr);
}
